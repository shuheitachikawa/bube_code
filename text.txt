1. クリップボードにコピー
  const copyText = "コピーしたい文字列";
  navigator.clipboard.writeText(copyText);

2. シェア画面を開く
  const title = "表示したいタイトル";
  const url = "シェアしたいURL";
  navigator.share({ title, url }) 

3. カメラを起動する
  navigator.mediaDevices.getUserMedia({video: true})

4. しゃべってもらう

  location.reload();
